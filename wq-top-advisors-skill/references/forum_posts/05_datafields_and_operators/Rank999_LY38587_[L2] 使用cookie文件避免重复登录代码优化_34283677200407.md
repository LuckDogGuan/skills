调试代码的时候需要反复login，其实使用cookie文件可以保存登录状态，只在cookie文件不存在或者失效的时候才真正登录

def relogin(username : str, password: str) -> requests.Session:

session = requests.Session()

session.auth = (username, password)

while True:

try:

response = session.post(" [https://api.worldquantbrain.com/authentication](https://api.worldquantbrain.com/authentication) ")

response.raise_for_status()

print("登录成功, cookie文件更新")

session.cookies = response.cookies

with open("cookie.json", "w") as f:

json.dump(requests.utils.dict_from_cookiejar(response.cookies), f)

return session

except requests.exceptions.RequestException as e:

print("登录失败，等待10s后再次尝试登录")

time.sleep(10)

return session

def login() -> requests.Session:

session = requests.Session()

if not os.path.exists("cookie.json"):

print("cookie文件不存在，直接使用账号密码登陆")

env_dist = os.environ

username = env_dist["WQ_USERNAME"]

password = env_dist["WQ_PASSWORD"]

session = relogin(username, password)

else:

with open("cookie.json", "r") as f:

cookies = requests.utils.cookiejar_from_dict(json.load(f))

session.cookies = cookies

response = session.get(" [https://api.worldquantbrain.com/operators](https://api.worldquantbrain.com/operators) ")

if response.status_code in (401, 403):

print("cookie文件失效，使用账号密码重新登陆")

env_dist = os.environ

username = env_dist["WQ_USERNAME"]

password = env_dist["WQ_PASSWORD"]

session = relogin(username, password)

else:

print("cookie文件有效")

return session
