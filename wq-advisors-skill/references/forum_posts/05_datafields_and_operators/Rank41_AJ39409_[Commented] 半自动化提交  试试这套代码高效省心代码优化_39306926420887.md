# 限流

elif res.status_code == 429:

def poll_status(s, alpha_id, submit_url, max_poll_time=2400):

"""

轮询 Alpha 验证状态

Token有效期4小时，只在401时刷新或每3小时定期刷新

"""

start_time = time.time()

count = 0

s_t = datetime.datetime.now()

base_interval = 30

last_refresh = time.time()

refresh_interval = 10800  # 3小时刷新一次Token（Token有效期4小时）

while time.time() - start_time < max_poll_time:

try:

# 每3小时定期刷新一次Token，或遇到401时刷新

if time.time() - last_refresh > refresh_interval:

print(f"🔄 定期刷新Token（已运行 {(time.time()-start_time)/60:.0f} 分钟）...")

s = refresh_login()

last_refresh = time.time()

res = s.get(submit_url, timeout=30)

# 还在处理中

if res.status_code == 200:

retry = res.headers.get('Retry-After', 0)

if retry:

count += 1

wait_time = float(retry)

# 每75次打印一次状态

if count % 75 == 0:

elapsed = datetime.datetime.now() - s_t

print(f"⏳ Alpha {alpha_id} GET 200. Waiting... {elapsed}.")

# ========== ✅ 修改点1：轮询间隔强制 ≥40秒 + 抖动 ==========

actual_wait = max(wait_time, 40.0)

time.sleep(actual_wait + random.uniform(2, 5))

else:

# 完成

print(f"✅ Alpha {alpha_id} submitted successfully.")

# 标记为黄色（成功）

safe_set_alpha_color(s, alpha_id, COLOR_YELLOW)

return 200

# 验证通过但质量不够 或 已提交

elif res.status_code == 403:

error_data = res.json()

checks = error_data.get("is", {}).get("checks", [])

# 检查是否是已提交状态

is_already_submitted = any(

check.get("name") == "ALREADY_SUBMITTED" and check.get("result") == "FAIL"

for check in checks

)

if is_already_submitted:

print(f"✅ Alpha {alpha_id} 已经提交成功 (ALREADY_SUBMITTED)")

# 标记为黄色（成功）

safe_set_alpha_color(s, alpha_id, COLOR_YELLOW)

return 200

else:

print(f"⚠️ Alpha {alpha_id} GET 403. Need Improvement.")

# 标记为红色（失败）

safe_set_alpha_color(s, alpha_id, COLOR_RED)

try:

print(pd.DataFrame(checks)[['name', 'value', 'result']])

except Exception:

try:

print(pd.DataFrame(checks)[['name', 'result']])

except:

pass

return 403

# 任务不存在或超时 - 不标记颜色，保留到下一轮

elif res.status_code == 404:

elapsed = datetime.datetime.now() - s_t

print(f"⏰ Alpha {alpha_id} GET 404. Time Out after {elapsed}.")

# 404是查询超时，不是提交失败，不标记颜色

return None

# 认证失败 - 刷新后继续

elif res.status_code == 401:

print(f"🔑 GET 401，刷新 Token...")

s = refresh_login()

last_refresh = time.time()

continue

# 限流

elif res.status_code == 429:

print(f"🚦 GET 429. Rate limited.")

# ========== ✅ 修改点4：429 等待至少 30 分钟 ==========

retry_after = int(res.headers.get('Retry-After', 1800))  # 默认 1800 秒（30分钟）

total_wait = max(retry_after, 1800) + random.randint(60, 180)

print(f"暂停 {total_wait} 秒（约 {total_wait//60} 分钟）...")

time.sleep(total_wait)

s = refresh_login()

last_refresh = time.time()

continue

# 其他错误

else:

print(f"⚠️ GET Status {res.status_code}.")

print(res.headers)

print(res.content)

return None

except requests.exceptions.Timeout:

print(f"⏱️ 轮询超时，继续...")

time.sleep(10)

except Exception as e:

print(f"❌ 轮询异常: {e}")

time.sleep(10)

print(f"⏰ 总体轮询超时 ({max_poll_time/60:.0f}分钟)")

return None
