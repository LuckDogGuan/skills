## 帖子正文

simulation中的settings字段很多都有级联关系，比如不同的universe有对应不同的neutralization，通过查看API： [https://api.worldquantbrain.com/users/{user_id}/alphas](https://api.worldquantbrain.com/users/{user_id}/alphas)  返回结果中的settings，找到了它们的对应关系：

1. region由instrumentType确定。
2. universe，delay和neutralization都是由instrumentType和region确定。

以下代码中的`alphas_settings.json`数据来自上面api返回的settings字段，这样保证了数据的正确性，避免手写可能引起的错误。后期如果平台字段有变动，大概率数据结构不会变。只用替换掉代码中的`alphas_settings.json`数据即可。

`get_different_settings`函数还会检查传入的参数是否符合级联关系，避免回测时报错。

```
from pathlib import Path
import json

settings_json_file = Path(__file__).parent / "alphas_settings.json"
with open(settings_json_file) as f:
    settings_data = json.load(f)

INSTRUMENT_TYPE_LIST = [item['value'] for item in settings_data['children']['instrumentType']['choices']]

def get_region_list(instrument_type):
    return [item['value'] for item in settings_data['children']['region']['choices']['instrumentType'][instrument_type] if item['value'] != 'NONE']

def get_universe_list(instrument_type, region):
    return [item['value'] for item in settings_data['children']['universe']['choices']['instrumentType'][instrument_type]['region'][region] if item['value'] != 'NONE']

def get_delay_list(instrument_type, region):
    return [item['value'] for item in settings_data['children']['delay']['choices']['instrumentType'][instrument_type]['region'][region] if item['value'] != 'NONE']

def get_neutralization_list(instrument_type, region):
    return [item['value'] for item in settings_data['children']['neutralization']['choices']['instrumentType'][instrument_type]['region'][region] if item['value'] != 'NONE']

lookback_days = [item['value'] for item in settings_data['children']['lookbackDays']['choices'] if item['value'] != 'NONE']

simulation_data = {
    "type": "REGULAR",
    "settings": '',
    "regular": ""
}

def dfs_values(values):
        from copy import copy
        ret = []
        tmp = []
        n = len(values)

        def dfs(i):
            if i == n:
                ret.append(copy(tmp))
            else:
                if isinstance(values[i], (list, tuple)):
                    for value in values[i]:
                        tmp.append(value)
                        dfs(i+1)
                        tmp.pop()
                else:
                    tmp.append(values[i])
                    dfs(i+1)
                    tmp.pop()
        dfs(0)
        return ret

def get_different_settings(instrumentType='EQUITY', region='USA', universe_list=None, truncation_list=None, decay_list=None, neutralization_list=None, delay_list=(1,)):

    if truncation_list is None:
        truncation_list = (0.08, )
    if decay_list is None:
        decay_list = list(range(0, 17, 2))
    assert region in get_region_list(instrumentType)
    o_neutralization_list = get_neutralization_list(instrumentType, region)
    if not neutralization_list:
        neutralization_list = o_neutralization_list
    else:
        assert all(neutralization in o_neutralization_list for neutralization in neutralization_list)
    o_delay_list = get_delay_list(instrumentType, region)
    if not delay_list:
        delay_list = o_delay_list
    else:
        assert all(delay in o_delay_list for delay in delay_list)

    o_universe_list = get_universe_list(instrumentType, region)
    if not universe_list:
        universe_list = o_universe_list
    else:
        assert all(universe in o_universe_list for universe in universe_list)

    settings = []
    for values in dfs_values([universe_list, truncation_list, decay_list, neutralization_list, delay_list]):
            values = dict(zip(['universe', 'truncation', 'decay', 'neutralization', 'delay'], values))
            settings.append({
                "nanHandling": "OFF",
                "instrumentType": instrumentType,
                "delay": values['delay'],
                "universe": values['universe'],
                "truncation": values['truncation'],
                "unitHandling": "VERIFY",
                "pasteurization": "ON",
                "region": region,
                "language": "FASTEXPR",
                "decay": values['decay'],
                "neutralization": values['neutralization'],
                "visualization": False
            })
    return settings

if __name__ == '__main__':
    a = get_different_settings(region='TWN', truncation_list=(0.01,))
    for settings in a:
        print(settings)
    print(len(a))
    for instrumentType in INSTRUMENT_TYPE_LIST:
        print(instrumentType + ':')
        for region in get_region_list(instrumentType):
            print(' ' + region)
            for universe in get_universe_list(instrumentType, region):
                print(' '*2 + universe, end='')
            print()
            for neutralization in get_neutralization_list(instrumentType, region):
                print(' '*2 + neutralization, end='')
            print()
            for delay in get_delay_list(instrumentType, region):
                print(' '*2 + str(delay), end='')
            print()
```
