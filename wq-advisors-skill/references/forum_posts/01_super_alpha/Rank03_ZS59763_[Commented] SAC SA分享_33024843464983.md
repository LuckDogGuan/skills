in(competitions, "HCAC2025")&&

not(own)&&

((turnover<0.03&&turnover>0.01)||

&&(operator_count<8)  &&(dataset_count<=2)

/(turnover*s_log_1p(turnover)

*sigmoid(self_correlation)

*sigmoid(prod_correlation)

*abs(long_count-short_count))
