def retry_request(method, url, max_retries=10, timeout=400, delay=20, **kwargs):

"""

通用的重试逻辑。

如果请求失败则进行重试，超时后重新登录。

如果重新登录超过 max_retries，则报错终止脚本。

:param method     : 请求方法 (e.g., SESS.get, SESS.patch)

:param url        : 请求的目标地址

:param max_retries: 最大重新登录次数

:param timeout    : 每次重试的超时时间

:param delay      : 两次请求的间隔时间

:param kwargs     : 请求参数 (e.g., json, headers)

"""

retry_count = 0

start_time  = time.time()

while retry_count <= max_retries:

try:

response = method(url, **kwargs)

response.raise_for_status()

result = response.json()

return result

except Exception as e:

elapsed_time = time.time() - start_time

if elapsed_time < timeout:

logging.warning(f"Request to {url} failed. Retrying after {delay} seconds... Elapsed time: {elapsed_time:.2f}s.")

time.sleep(delay)

else:

if retry_count < max_retries:

logging.warning(f"Timeout reached. Attempting re-login... (Retry count: {retry_count + 1})")

global SESS

SESS = global_sign_in()

start_time = time.time()

retry_count += 1

else:

logging.error(f"Request to {url} failed after {max_retries} re-login attempts.")

return None

def check_alpha_submission(alpha_id):

base_url = f" [https://api.worldquantbrain.com/alphas/{alpha_id}](https://api.worldquantbrain.com/alphas/{alpha_id}) "

# 1. 标记为黄色，触发重新检查

if retry_request(SESS.patch, base_url, json={"color": "YELLOW"}):

logging.info(f"Mark Alpha {alpha_id} in YELLOW, Reset SELF_CORRELATION to PENDING.")

else:

logging.error(f"Mark Alpha {alpha_id} in YELLOW Fail, Skip this Alpha.")

return False

# 2. 获取检查结果

result = retry_request(SESS.get, f"{base_url}/check", delay=60)

if result is None:

logging.error(f"Get Alpha {alpha_id} IS CHECK Status Fail, Skip this Alpha.")

return False

# 3. 打印 API 返回的原始数据（调试用）

# logging.debug(f"Raw API response for Alpha {alpha_id}:")

# logging.debug(json.dumps(result, indent=2))  # 格式化 JSON 输出

# 4. 解析检查项

checks = result["is"]["checks"]

if all(check.get("result") in ("PASS","WARNING") for check in checks):

if retry_request(SESS.patch, base_url, json={"color": "GREEN"}):

logging.info(f"Alpha {alpha_id} Check Submission PASS, Mark Alpha in GREEN.")

else:

logging.warning(f"Alpha {alpha_id} Check Submission PASS, But Mark Alpha in GREEN Fail.")

return True

else:

if retry_request(SESS.patch, base_url, json={"color": "RED"}):

logging.info(f"Alpha {alpha_id} Check Submission FAIL, MARK Alpha in RED.")

else:

logging.warning(f"Alpha {alpha_id} Check Submission FAIL, MARK Alpha in RED Fail.")

return False

def get_success_alphas(alpha_list):

alpha_success_list  = []

total_alpha_checke  = len(alpha_list)

alpha_succeed_count = 0

csv_filename = "alpha_success_check.csv"

csv_columns = ["id", "status", "dateCreated", "is"]

# 先写入 CSV 头

with open(csv_filename, "w", newline="", encoding="utf-8") as csvfile:

writer = csv.DictWriter(csvfile, fieldnames=csv_columns, extrasaction="ignore")

writer.writeheader()

# 使用 ThreadPoolExecutor 来并行调用 check_alpha_submission

with ThreadPoolExecutor(max_workers=5) as executor:  # 最大线程数为 2，可以根据实际需求调整

futures = {}

for index, alpha in enumerate(alpha_list, start=1):

alpha_id = alpha.get("id")

futures[executor.submit(check_alpha_submission, alpha_id)] = (index, alpha)

logging.info(f"+[{index:04d}/{total_alpha_checke:04d}] Alpha id: {alpha_id} Start Check Submission" + "-"*72 + "+")

# 遍历并收集线程的执行结果

for future in as_completed(futures):

index, alpha = futures[future]

try:

result = future.result()

if result:

alpha_succeed_count += 1

alpha_success_list.append(alpha)

# 追加写入 CSV（实时更新）

with open(csv_filename, "a", newline="", encoding="utf-8") as csvfile:

writer = csv.DictWriter(csvfile, fieldnames=csv_columns, extrasaction="ignore")

writer.writerow(alpha)

except Exception as e:

logging.error(f"Error while checking Alpha {alpha.get('id')}: {e}")

logging.info(f"Currently {alpha_succeed_count} Alphas PASS Check Submission.")

logging.info("+" + "-"*124 + "+")

return alpha_success_list

return alpha_final_list

def retry_request(method, url, max_retries=10, timeout=400, delay=20, **kwargs):

"""

通用的重试逻辑。

如果请求失败则进行重试，超时后重新登录。

如果重新登录超过 max_retries，则报错终止脚本。

:param method     : 请求方法 (e.g., SESS.get, SESS.patch)

:param url        : 请求的目标地址

:param max_retries: 最大重新登录次数

:param timeout    : 每次重试的超时时间

:param delay      : 两次请求的间隔时间

:param kwargs     : 请求参数 (e.g., json, headers)

"""

retry_count = 0

start_time  = time.time()

while retry_count <= max_retries:

try:

response = method(url, **kwargs)

response.raise_for_status()

result = response.json()

return result

except Exception as e:

elapsed_time = time.time() - start_time

if elapsed_time < timeout:

logging.warning(f"Request to {url} failed. Retrying after {delay} seconds... Elapsed time: {elapsed_time:.2f}s.")

time.sleep(delay)

else:

if retry_count < max_retries:

logging.warning(f"Timeout reached. Attempting re-login... (Retry count: {retry_count + 1})")

global SESS

SESS = global_sign_in()

start_time = time.time()

retry_count += 1

else:

logging.error(f"Request to {url} failed after {max_retries} re-login attempts.")

return None

def check_alpha_submission(alpha_id):

base_url = f" [https://api.worldquantbrain.com/alphas/{alpha_id}](https://api.worldquantbrain.com/alphas/{alpha_id}) "

# 1. 标记为黄色，触发重新检查

if retry_request(SESS.patch, base_url, json={"color": "YELLOW"}):

logging.info(f"Mark Alpha {alpha_id} in YELLOW, Reset SELF_CORRELATION to PENDING.")

else:

logging.error(f"Mark Alpha {alpha_id} in YELLOW Fail, Skip this Alpha.")

return False

# 2. 获取检查结果

result = retry_request(SESS.get, f"{base_url}/check", delay=60)

if result is None:

logging.error(f"Get Alpha {alpha_id} IS CHECK Status Fail, Skip this Alpha.")

return False

# 3. 打印 API 返回的原始数据（调试用）

# logging.debug(f"Raw API response for Alpha {alpha_id}:")

# logging.debug(json.dumps(result, indent=2))  # 格式化 JSON 输出

# 4. 解析检查项

checks = result["is"]["checks"]

if all(check.get("result") in ("PASS","WARNING") for check in checks):

if retry_request(SESS.patch, base_url, json={"color": "GREEN"}):

logging.info(f"Alpha {alpha_id} Check Submission PASS, Mark Alpha in GREEN.")

else:

logging.warning(f"Alpha {alpha_id} Check Submission PASS, But Mark Alpha in GREEN Fail.")

return True

else:

if retry_request(SESS.patch, base_url, json={"color": "RED"}):

logging.info(f"Alpha {alpha_id} Check Submission FAIL, MARK Alpha in RED.")

else:

logging.warning(f"Alpha {alpha_id} Check Submission FAIL, MARK Alpha in RED Fail.")

return False

def get_success_alphas(alpha_list):

alpha_success_list  = []

total_alpha_checke  = len(alpha_list)

alpha_succeed_count = 0

csv_filename = "alpha_success_check.csv"

csv_columns = ["id", "status", "dateCreated", "is"]

# 先写入 CSV 头

with open(csv_filename, "w", newline="", encoding="utf-8") as csvfile:

writer = csv.DictWriter(csvfile, fieldnames=csv_columns, extrasaction="ignore")

writer.writeheader()

# 使用 ThreadPoolExecutor 来并行调用 check_alpha_submission

with ThreadPoolExecutor(max_workers=5) as executor:  # 最大线程数为 2，可以根据实际需求调整

futures = {}

for index, alpha in enumerate(alpha_list, start=1):

alpha_id = alpha.get("id")

futures[executor.submit(check_alpha_submission, alpha_id)] = (index, alpha)

logging.info(f"+[{index:04d}/{total_alpha_checke:04d}] Alpha id: {alpha_id} Start Check Submission" + "-"*72 + "+")

# 遍历并收集线程的执行结果

for future in as_completed(futures):

index, alpha = futures[future]

try:

result = future.result()

if result:

alpha_succeed_count += 1

alpha_success_list.append(alpha)

# 追加写入 CSV（实时更新）

with open(csv_filename, "a", newline="", encoding="utf-8") as csvfile:

writer = csv.DictWriter(csvfile, fieldnames=csv_columns, extrasaction="ignore")

writer.writerow(alpha)

except Exception as e:

logging.error(f"Error while checking Alpha {alpha.get('id')}: {e}")

logging.info(f"Currently {alpha_succeed_count} Alphas PASS Check Submission.")

logging.info("+" + "-"*124 + "+")

return alpha_success_list

def get_checked_alphas(alpha_list, max_alphas=None):

alpha_checked_list = []

for alpha in alpha_list:

# 如果已达到 max_alphas 限制，提前终止循环

if max_alphas is not None and len(alpha_checked_list) >= max_alphas:

break

# 获取 Alpha 的检查结果

is_checks = alpha.get("is", {}).get("checks", [])

# 确保没有 'FAIL' 项

if all(check.get("result") != "FAIL" for check in is_checks):

alpha_checked_list.append(alpha)

logging.info("+" + "="*124 + "+")

logging.info(f"{len(alpha_list)} Alphas Passed Filter (Sharpe>1.58, Fitness>1, 70>Turnover>1).")

logging.info(f"{len(alpha_checked_list)} Alphas IS CHECKS No FAIL, Waiting to Check Submission")

logging.info("+" + "="*124 + "+")

print(f"{len(alpha_checked_list) = }")

# 写入 CSV

csv_filename = "alpha_pending_check.csv"

csv_columns = ["id", "status", "dateCreated", "is"]

with open(csv_filename, "w", newline="", encoding="utf-8") as csvfile:

writer = csv.DictWriter(

csvfile,

fieldnames=csv_columns,

extrasaction="ignore"

)

writer.writeheader()

# 写入筛选后的列表

writer.writerows(alpha_checked_list[:max_alphas] if max_alphas else alpha_checked_list)

# 返回筛选后的列表

return alpha_checked_list[:max_alphas] if max_alphas else alpha_checked_list
