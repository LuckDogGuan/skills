## 帖子正文

需求背景，在使用wqb生成json文件后，回测的时候遇到以下两个问题：
1、每一次都需要重新的选择文件，然后点击回测按钮，虽然wqb已经界面化了，但有时候需要标记哪些文件已经回测完毕，哪些没有。

2、不知道什么时候回测完成，需要不定时的登录云服务器查看是否回测结束，比较麻烦。

针对上面的问题，能否使用程序自动化的扫描指定的文件目录，然后逐个回测文件里的alpha表达式，完成之后继续处理下一个文件，不需要手动的去回测某一个文件，在处理每个文件的时候，使用多线程来处理回测，以下是完整的代码：

import os
import json
import shutil
import time
import threading
from queue import Queue
import requests
from requests.exceptions import HTTPError
import logging
from logging.handlers import RotatingFileHandler
from datetime import datetime

# 配置参数
INPUT_DIR = "待处理文件目录"  # 替换为实际输入目录
DONE_DIR = "已完成文件目录"   # 替换为实际完成目录
LOG_DIR = "处理日志目录"       # 日志文件存储目录
NODE_THREAD_COUNT = 4          # 每个文件内部处理节点的线程数量
API_URL = " [https://api.worldquantbrain.com/simulations](https://api.worldquantbrain.com/simulations) "
LOGIN_URL = " [https://api.worldquantbrain.com/login](https://api.worldquantbrain.com/login) "
SCAN_INTERVAL = 60             # 没有文件时，扫描间隔时间（秒）
RETRY_ATTEMPTS = 3             # 单个节点处理的重试次数
RETRY_DELAY = 60               # 重试延迟时间（秒）
MAX_LOG_SIZE = 10 * 1024 * 1024  # 最大日志文件大小（10MB）
BACKUP_LOG_COUNT = 5            # 备份日志文件数量

# 确保目录存在
os.makedirs(INPUT_DIR, exist_ok=True)
os.makedirs(DONE_DIR, exist_ok=True)
os.makedirs(LOG_DIR, exist_ok=True)

# 配置日志
def setup_logger():
    """配置日志系统，同时输出到控制台和文件"""
    logger = logging.getLogger('file_processor')
    logger.setLevel(logging.INFO)

    # 避免重复设置处理器
    if logger.handlers:
        return logger

    # 日志格式
    formatter = logging.Formatter(
        '%(asctime)s - %(levelname)s - %(message)s',
        datefmt='%Y-%m-%d %H:%M:%S'
    )

    # 控制台处理器
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(formatter)
    logger.addHandler(console_handler)

    # 文件处理器（支持日志轮转）
    log_filename = os.path.join(LOG_DIR, f"processor_{datetime.now().strftime('%Y%m%d')}.log")
    file_handler = RotatingFileHandler(
        log_filename,
        maxBytes=MAX_LOG_SIZE,
        backupCount=BACKUP_LOG_COUNT,
        encoding='utf-8'
    )
    file_handler.setFormatter(formatter)
    logger.addHandler(file_handler)

    return logger

# 初始化日志
logger = setup_logger()

# 全局会话锁，确保登录操作线程安全
session_lock = threading.Lock()

def login():
    """获取认证会话，返回新会话"""
    try:
        session = requests.Session()
        # 完善登录逻辑，示例：
        # response = session.post(LOGIN_URL, json={"username": "your_username", "password": "your_password"})
        # response.raise_for_status()  # 确保登录成功
        logger.info("成功登录并获取新会话")
        return session
    except Exception as e:
        logger.error(f"登录失败: {str(e)}")
        # 登录失败时等待一段时间后重试
        time.sleep(RETRY_DELAY)
        return login()

def process_single_node(session, node, node_id, file_name):
    """处理单个节点，仅在捕获401错误时重新登录"""
    current_session = session
    log_prefix = f"文件 {file_name} 节点 {node_id}"

    for attempt in range(RETRY_ATTEMPTS):
        try:            
            sim_data = {
                "type": node["type"],
                "settings": node["settings"],
                "regular": node["regular"]
            }

            # 发送模拟请求
            response = current_session.post(API_URL, json=sim_data)
            response.raise_for_status()  # 触发HTTP错误（包括401）

            # 获取进度URL
            progress_url = response.headers.get("Location")
            if not progress_url:
                logger.warning(f"{log_prefix} 未获取到进度URL")
                return False, current_session

# 轮询等待完成
            while True:
                progress_response = current_session.get(progress_url)
                progress_response.raise_for_status()  # 检查HTTP错误

                retry_after = progress_response.headers.get("Retry-After", 0)
                if retry_after == 0:
                    break
                time.sleep(float(retry_after))

# 检查状态
            result = progress_response.json()
            if result.get("status") != "COMPLETE":
                logger.warning(f"{log_prefix} 处理未完成")
                return False, current_session

            logger.info(f"{log_prefix} 处理完成")
            return True, current_session

except HTTPError as e:
            if e.response.status_code == 401:
                logger.warning(f"{log_prefix} 第 {attempt+1} 次尝试失败: 会话过期 (401)")
                # 加锁确保登录操作线程安全
                with session_lock:
                    current_session = login()
            else:
                logger.error(f"{log_prefix} 第 {attempt+1} 次尝试失败: HTTP错误 {e.response.status_code}")
        except Exception as e:
            logger.error(f"{log_prefix} 第 {attempt+1} 次尝试失败: {str(e)}")

        # 如果不是最后一次尝试，等待后重试
        if attempt < RETRY_ATTEMPTS - 1:
            logger.info(f"{log_prefix} 将在 {RETRY_DELAY} 秒后重试...")
            time.sleep(RETRY_DELAY)

    logger.error(f"{log_prefix} 达到最大重试次数，处理失败")
    return False, current_session

def process_file(file_path, session):
    """处理单个文件，内部使用多线程处理节点"""
    file_name = os.path.basename(file_path)
    logger.info(f"开始处理文件: {file_name}")
    start_time = time.time()
    success_count = 0
    total_nodes = 0

    try:
        # 读取文件内容
        with open(file_path, 'r', encoding='utf-8') as f:
            nodes = json.load(f)

        total_nodes = len(nodes)
        if not nodes:
            logger.info(f"文件 {file_name} 内容为空，跳过处理")
            return True, session

        # 创建节点处理队列
        node_queue = Queue()
        for i, node in enumerate(nodes):
            node_queue.put((i + 1, node))  # (节点ID, 节点数据)

        # 用于线程间共享最新会话的变量
        shared_session = [session]
        # 用于计数成功处理的节点
        success_counter = [0]

        # 节点处理线程函数
        def node_worker():
            thread_session = shared_session[0]
            while not node_queue.empty():
                node_id, node = node_queue.get()
                try:
                    success, new_session = process_single_node(thread_session, node, node_id, file_name)
                    if success:
                        with threading.Lock():
                            success_counter[0] += 1
                    # 同步更新会话
                    if new_session is not thread_session:
                        with session_lock:
                            shared_session[0] = new_session
                            thread_session = new_session
                finally:
                    node_queue.task_done()

        # 启动节点处理线程
        node_threads = []
        thread_count = min(NODE_THREAD_COUNT, len(nodes))
        for _ in range(thread_count):
            t = threading.Thread(target=node_worker)
            t.start()
            node_threads.append(t)

        # 等待所有节点处理完成
        node_queue.join()

        # 等待所有线程结束
        for t in node_threads:
            t.join()

        success_count = success_counter[0]
        elapsed_time = time.time() - start_time
        logger.info(
            f"文件 {file_name} 处理完成 - "
            f"总节点数: {total_nodes}, "
            f"成功处理: {success_count}, "
            f"耗时: {elapsed_time:.2f}秒"
        )
        return True, shared_session[0]

    except Exception as e:
        logger.error(f"文件 {file_name} 处理出错: {str(e)}", exc_info=True)
        return False, session

def check_and_process_files(session):
    """检查并处理目录中的文件，返回是否有文件被处理及更新后的会话"""
    # 获取所有待处理文件
    files = [
        os.path.join(INPUT_DIR, f) 
        for f in os.listdir(INPUT_DIR) 
        if os.path.isfile(os.path.join(INPUT_DIR, f))
    ]

    if not files:
        logger.info("当前没有待处理文件")
        return False, session  # 没有文件被处理

    current_session = session
    processed_count = 0
    success_count = 0

    logger.info(f"发现 {len(files)} 个待处理文件，开始处理...")

    # 逐个处理文件
    for file_path in files:
        processed_count += 1
        file_name = os.path.basename(file_path)
        logger.info(f"开始处理第 {processed_count}/{len(files)} 个文件: {file_name}")

        # 处理当前文件
        success, current_session = process_file(file_path, current_session)

        # 如果处理成功，移动文件到已完成目录
        if success:
            success_count += 1
            dest_path = os.path.join(DONE_DIR, file_name)
            shutil.move(file_path, dest_path)
            logger.info(f"文件 {file_name} 已移动到完成目录")
        else:
            logger.warning(f"文件 {file_name} 处理失败，将保留在原目录")

    logger.info(f"本轮文件处理完成 - 总文件数: {len(files)}, 成功处理: {success_count}")
    return True, current_session  # 有文件被处理

def main():
    logger.info("=" * 50)
    logger.info("开始持续文件处理服务...")
    logger.info(f"监控目录: {os.path.abspath(INPUT_DIR)}")
    logger.info(f"完成目录: {os.path.abspath(DONE_DIR)}")
    logger.info(f"日志目录: {os.path.abspath(LOG_DIR)}")
    logger.info(f"扫描间隔: {SCAN_INTERVAL}秒")
    logger.info(f"节点处理线程数: {NODE_THREAD_COUNT}")
    logger.info("=" * 50)

    # 初始登录
    session = login()

    try:
        while True:
            # 检查并处理文件
            has_processed, session = check_and_process_files(session)

            if not has_processed:
                # 没有文件可处理，等待一段时间后再次扫描
                logger.info(f"将在 {SCAN_INTERVAL} 秒后再次扫描目录...")
                time.sleep(SCAN_INTERVAL)

    except KeyboardInterrupt:
        logger.info("\n用户中断，程序退出")
    except Exception as e:
        logger.critical(f"程序发生致命错误: {str(e)}", exc_info=True)
    finally:
        logger.info("文件处理服务已停止")
        logger.info("=" * 50)

if __name__ == "__main__":
    main()
