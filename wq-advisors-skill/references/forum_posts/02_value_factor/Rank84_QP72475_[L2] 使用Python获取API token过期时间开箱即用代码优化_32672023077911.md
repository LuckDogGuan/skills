**工具函数代码:**

```
import base64import jsonfrom datetime import datetimedef decode_jwt_exp(jwt_token):    try:        # 分割 JWT 的三部分        parts = jwt_token.split('.')        if len(parts) != 3:            raise ValueError("这不是一个有效的 JWT")        # Base64Url 解码 Payload        payload_b64 = parts[1]        padding = '=' * (-len(payload_b64) % 4)  # 补全 Base64        payload_json = base64.urlsafe_b64decode(payload_b64 + padding).decode('utf-8')        payload = json.loads(payload_json)        # 提取 exp 字段        exp_timestamp = payload.get('exp')        if not exp_timestamp:            print("未找到 exp 字段")            return        # 转换时间        exp_time = datetime.fromtimestamp(exp_timestamp)          print("过期时间（UTC）:", exp_time.strftime('%Y-%m-%d %H:%M:%S'))    except Exception as e:        print("解码出错:", e)
```

1.网页微调判断是否重新登陆刷新一下token
