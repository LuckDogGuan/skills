2. 提示输入 scope（比如 GLB/D0），自动爬取该 scope 当前所有 OS 分数
3. 列出当前分数，5 秒自动确认（输入 n 取消）
4. 按原始比例重新分配为恰好 100,000 分
5. 一个一个顺序写回平台，碰到 429 等 5 秒重试，最多 5 次
6. 完成后继续等待下一个 scope，直接回车退出

# -*- coding: utf-8 -*-
"""
《即插即用》OS 自动重分配到恰好 10 万分

交互式运行，无需命令行参数。启动后会引导你填写 scope，改完一个接着改下一个。
首次运行时如果没有 brain.txt，会引导你输入邮箱密码并自动保存。

退出方式：scope 为空直接回车，或 Ctrl+C。
"""

from __future__ import annotations

import json
import os
import sys

import time
from datetime import datetime
from pathlib import Path
from typing import Any, Optional

import pandas as pd
import requests

# ── 常量 ────────────────────────────────────────────────────────────────
SCRIPT_DIR = Path(__file__).resolve().parent
REPO_ROOT = SCRIPT_DIR.parent
BASE_URL = " [https://api.worldquantbrain.com](https://api.worldquantbrain.com) "
DEFAULT_TOTAL_POINTS = 100_000
DEFAULT_RETRIES = 5
DEFAULT_RETRY_DELAY = 5.0

# ── 工具函数 ────────────────────────────────────────────────────────────
def now_stamp() -> str:
    return datetime.now().strftime("%Y%m%d_%H%M%S")

def _json_ready(value: Any) -> Any:
    """将 numpy/pandas 类型转为 JSON 可序列化类型。"""
    import numpy as np
    if isinstance(value, (np.integer,)):
        return int(value)
    if isinstance(value, (np.floating,)):
        return None if np.isnan(value) else float(value)
    if isinstance(value, pd.Timestamp):
        return value.isoformat()
    if pd.isna(value):
        return None
    return value

def format_scope(region: Any, delay: Any) -> str:
    region_text = "" if pd.isna(region) else str(region).strip().upper()
    delay_text = "" if pd.isna(delay) else str(delay).strip()
    return f"{region_text}/D{delay_text}" if region_text and delay_text else ""

def _parse_scope(scope: str) -> tuple[str, str]:
    scope_text = str(scope).strip().upper()
    if "/D" not in scope_text:
        raise ValueError(f"无效的 scope 格式: {scope}，应为 REGION/Dn 如 GLB/D0")
    region, delay = scope_text.split("/D", 1)
    if not region or delay == "":
        raise ValueError(f"无效的 scope 格式: {scope}")
    return region, delay

# ── 认证 ────────────────────────────────────────────────────────────────
def build_authenticated_session(credentials_path: str) -> requests.Session:
    cred_path = Path(credentials_path).expanduser().resolve()
    username, password = json.loads(cred_path.read_text(encoding="utf-8"))
    session = requests.Session()
    session.auth = (username, password)

last_error: Optional[Exception] = None
    for attempt in range(5):
        try:
            response = session.post(f"{BASE_URL}/authentication", json={}, timeout=30)
            response.raise_for_status()
            return session
        except requests.HTTPError as exc:
            last_error = exc
            status_code = exc.response.status_code if exc.response is not None else None
            if status_code != 429 or attempt >= 4:
                raise
            time.sleep(1.0 * (attempt + 1))
        except Exception as exc:
            last_error = exc
            if attempt >= 4:
                raise
            time.sleep(1.0 * (attempt + 1))
    raise last_error  # type: ignore[misc]

# ── 平台 API ────────────────────────────────────────────────────────────
def _flatten_alpha_record(record: dict[str, Any]) -> dict[str, Any]:
    settings = record.get("settings") or {}
    return {
        "alpha_id": record.get("id"),
        "type": record.get("type"),
        "stage": record.get("stage"),
        "status": record.get("status"),
        "settings_region": settings.get("region"),
        "settings_delay": settings.get("delay"),
        "osmosisPoints": record.get("osmosisPoints"),
        "scope": format_scope(settings.get("region"), settings.get("delay")),
    }

def fetch_current_scope_allocations(
    scope: str,
    *,
    credentials_path: str | None = None,
) -> tuple[pd.DataFrame, dict[str, Any]]:
    """从平台获取指定 scope 当前所有带 Osmosis 分数的 alpha。"""
    if credentials_path is None:
        credentials_path = str(REPO_ROOT / "brain.txt")
    region, delay = _parse_scope(scope)
    session = build_authenticated_session(credentials_path)

rows: list[dict[str, Any]] = []
    limit = 100
    offset = 0

while True:
        url = (
            f"{BASE_URL}/users/self/alphas?limit={limit}&offset={offset}"
            f"&status!=UNSUBMITTED%1FIS-FAIL"
            f"&settings.region={region}&settings.delay={delay}"
            f"&osmosisPoints%3E=0&order=-dateSubmitted&hidden=false"
        )
        response = session.get(url, timeout=60)
        response.raise_for_status()
        payload = response.json()
        results = payload.get("results") or []
        rows.extend(_flatten_alpha_record(item) for item in results)

next_url = payload.get("next")
        if not next_url:
            break
        offset += limit
        if len(results) < limit:
            break

df = pd.DataFrame(rows)
    if df.empty:
        df = pd.DataFrame(
            columns=["alpha_id", "type", "stage", "status",
                     "settings_region", "settings_delay", "osmosisPoints", "scope"]
        )
    df["scope"] = df.apply(
        lambda row: format_scope(row.get("settings_region"), row.get("settings_delay")),
        axis=1,
    )
    df["osmosisPoints"] = pd.to_numeric(df["osmosisPoints"], errors="coerce")
    df = df[(df["scope"] == str(scope).upper()) & (df["osmosisPoints"].notna())].copy()

meta = {
        "scope": str(scope).upper(),
        "active_alpha_count": int(len(df)),
        "active_point_sum": _json_ready(df["osmosisPoints"].sum()) if not df.empty else 0,
    }
    return df, meta

def patch_alpha_osmosis_points(
    alpha_id: str,
    osmosis_points: Optional[int],
    session: requests.Session,
) -> dict[str, Any]:
    """写入单个 alpha 的 Osmosis 分数，返回确认结果。"""
    payload = {"osmosisPoints": None if osmosis_points is None else int(osmosis_points)}
    response = session.patch(f"{BASE_URL}/alphas/{alpha_id}", json=payload, timeout=60)
    response.raise_for_status()
    # PATCH 响应通常直接返回更新后的 alpha 对象
    patched = response.json()
    return {
        "alpha_id": alpha_id,
        "requested_osmosisPoints": payload["osmosisPoints"],
        "confirmed_osmosisPoints": patched.get("osmosisPoints"),
        "status": "ok",
        "scope": format_scope(
            (patched.get("settings") or {}).get("region"),
            (patched.get("settings") or {}).get("delay"),
        ),
        "type": patched.get("type"),
        "stage": patched.get("stage"),
    }

# ── 重试包装 ────────────────────────────────────────────────────────────
def patch_with_retry(
    alpha_id: str,
    points: Optional[int],
    session: requests.Session,
    *,
    max_retries: int = DEFAULT_RETRIES,
    delay: float = DEFAULT_RETRY_DELAY,
) -> dict[str, Any]:
    """带重试的 patch 操作。遇到任何异常都会重试。"""
    last_error: Optional[Exception] = None
    for attempt in range(max_retries + 1):
        try:
            return patch_alpha_osmosis_points(alpha_id, points, session)
        except Exception as exc:
            last_error = exc
            if attempt < max_retries:
                print(f"\n  ⚠ {alpha_id} 失败 (第 {attempt+1}/{max_retries+1} 次): {exc}")
                print(f"     {delay}s 后重试...", end="", flush=True)
                time.sleep(delay)
    # 所有重试耗尽
    return {
        "alpha_id": alpha_id,
        "requested_osmosisPoints": points,
        "confirmed_osmosisPoints": None,
        "status": "error",
        "error": str(last_error),
    }

# ── 最大余数法分配 ──────────────────────────────────────────────────────
def largest_remainder_allocate(
    weights: list[float],
    total: int = DEFAULT_TOTAL_POINTS,
) -> list[int]:
    """按权重比例分配整数，保证总和正好等于 total（最大余数法）。"""
    if not weights:
        return []
    weight_sum = sum(weights)
    if weight_sum <= 0:
        n = len(weights)
        base = total // n
        result = [base] * n
        result[-1] += total - base * n
        return result

n = len(weights)
    quotas = [w / weight_sum * total for w in weights]
    floors = [int(q) for q in quotas]
    remainders = [quotas[i] - floors[i] for i in range(n)]

allocated = sum(floors)
    remaining = total - allocated

order = sorted(range(n), key=lambda i: remainders[i], reverse=True)
    for i in range(remaining):
        floors[order[i]] += 1

return floors

# ── 确认交互 ────────────────────────────────────────────────────────────
def confirm_with_timeout(prompt: str, timeout: float = 5.0) -> bool:
    """等待用户确认，超时自动返回 True。

按键即响应，无需回车。
      n / N → 取消
      其他任意键 → 确认
      超时 → 自动确认
    """
    import msvcrt
    print(prompt, end="", flush=True)

for remaining in range(int(timeout), 0, -1):
        print(f" {remaining}s ", end="", flush=True)
        tick = time.time()
        while time.time() - tick < 0.5:
            if msvcrt.kbhit():
                key = msvcrt.getch()
                try:
                    ch = key.decode("utf-8").lower()
                except UnicodeDecodeError:
                    ch = ""
                if ch == "n":
                    print()
                    return False
                print()
                return True
        print("\b" * 4, end="", flush=True)  # 退回4格覆盖 " Xs "

print("⏱ 自动确认")
    return True

# ── 主流程 ──────────────────────────────────────────────────────────────
# ── 认证文件管理 ────────────────────────────────────────────────────────
def _resolve_credentials_path() -> str:
    """查找或创建 brain.txt。

优先级: SCRIPT_DIR/brain.txt > REPO_ROOT/brain.txt > 引导用户输入。
    首次输入后自动保存到 SCRIPT_DIR。
    """
    candidates = [
        SCRIPT_DIR / "brain.txt",
        REPO_ROOT / "brain.txt",
    ]
    for p in candidates:
        if p.exists():
            return str(p)

print("🔐 未找到 brain.txt，请输入 Brain 平台认证信息：")
    print()
    email = input("  邮箱: ").strip()
    pwd = input("  密码: ").strip()
    if not email or not pwd:
        print("❌ 邮箱或密码为空，退出。")
        sys.exit(1)

save_path = SCRIPT_DIR / "brain.txt"
    save_path.write_text(json.dumps([email, pwd], ensure_ascii=False), encoding="utf-8")
    print(f"  ✅ 已保存到 {save_path}")
    print()
    return str(save_path)

# ── 单次 scope 处理 ────────────────────────────────────────────────────
def process_scope(scope: str, credentials_path: str) -> None:
    """对一个 scope 执行完整的爬取→展示→确认→分配→写入流程。"""
    print(f"目标总分: {DEFAULT_TOTAL_POINTS:,}")
    print()

# ── 1. 爬取 ──────────────────────────────────────────────────
    print("📡 正在从平台爬取当前 Osmosis 打分 ...")
    try:
        df, meta = fetch_current_scope_allocations(scope, credentials_path=credentials_path)
    except Exception as exc:
        print(f"❌ 爬取失败: {exc}")
        return

if df.empty:
        print(f"⚠ Scope {scope} 当前没有有效的 Osmosis 分配。")
        return

raw_total = int(df["osmosisPoints"].sum())
    print(f"✅ 爬取完成: {meta['active_alpha_count']} 个 alpha，当前总和 {raw_total:,}")
    print()

# ── 2. 展示 ──────────────────────────────────────────────────
    display = df[["alpha_id", "type", "osmosisPoints"]].copy()
    display = display.sort_values("osmosisPoints", ascending=False)
    display["pct"] = display["osmosisPoints"] / raw_total * 100 if raw_total > 0 else 0

print("── 平台当前 Osmosis 分数 ──")
    for _, row in display.iterrows():
        aid = row["alpha_id"]
        pts = int(row["osmosisPoints"])
        pct = row["pct"]
        t = row.get("type", "")
        print(f"  {aid:<16s}  {t:<8s}  {pts:>6d}  ({pct:5.1f}%)")
    print(f"  合计: {raw_total:,}")
    print()

# ── 3. 确认 ──────────────────────────────────────────────────
    if raw_total == DEFAULT_TOTAL_POINTS:
        print(f"✨ 当前总和已等于 {DEFAULT_TOTAL_POINTS:,}，无需操作。")
        return

print(f"⚠ 当前总和 {raw_total:,} ≠ {DEFAULT_TOTAL_POINTS:,}，将按比例重新分配。")
    print(f"   等待 5s 自动确认（按 n 取消，其他任意键直接确认）")
    print()

try:
        confirmed = confirm_with_timeout("确认重新分配？[Y/n] ", timeout=5.0)
    except KeyboardInterrupt:
        print("\n已取消。")
        return

if not confirmed:
        print("已取消。")
        return
    print()

# ── 4. 重新分配 ──────────────────────────────────────────────
    weights = df["osmosisPoints"].tolist()
    new_points = largest_remainder_allocate(weights, DEFAULT_TOTAL_POINTS)

new_df = df.copy()
    new_df["new_points"] = new_points

print("── 新旧对比 ──")
    header = f"  {'Alpha ID':<16s}  {'类型':<8s}  {'旧分':>6s}  {'新分':>6s}  {'旧%':>5s}  {'新%':>5s}  {'变化':>5s}"
    print(header)
    print("  " + "-" * 68)
    compare = new_df[["alpha_id", "type", "osmosisPoints", "new_points"]].copy()
    compare = compare.sort_values("new_points", ascending=False)
    for _, row in compare.iterrows():
        aid = row["alpha_id"]
        t = row.get("type", "")
        old = int(row["osmosisPoints"])
        new = int(row["new_points"])
        old_pct = old / raw_total * 100 if raw_total > 0 else 0
        new_pct = new / DEFAULT_TOTAL_POINTS * 100
        diff = new - old
        diff_s = f"+{diff}" if diff > 0 else str(diff)
        print(f"  {aid:<16s}  {t:<8s}  {old:>6d}  {new:>6d}  {old_pct:>4.1f}%  {new_pct:>4.1f}%  {diff_s:>5s}")
    print()
    print(f"  旧总和: {raw_total:,}  →  新总和: {sum(new_points):,}")
    print()

# ── 5. 写入 ──────────────────────────────────────────────────
    ops: list[tuple[str, Optional[int]]] = []

for _, row in new_df.iterrows():
        aid = str(row["alpha_id"])
        pts = int(row["new_points"])
        old_pts = int(row.get("osmosisPoints", 0))
        if old_pts == pts:
            print(f"  ⏭ {aid} 已是 {pts}，跳过")
            continue
        ops.append((aid, pts))

if not ops:
        print("✅ 无需任何操作。")
        return

print(f"📤 准备写入 {len(ops)} 个操作 ...")
    print()

session = build_authenticated_session(credentials_path)
    ok_count = 0
    err_count = 0

for i, (aid, pts) in enumerate(ops, 1):
        print(f"  [{i}/{len(ops)}] {aid} → {pts} ...", end="", flush=True)

result = patch_with_retry(
            aid, pts, session,
            max_retries=DEFAULT_RETRIES,
            delay=DEFAULT_RETRY_DELAY,
        )

if result["status"] == "ok":
            ok_count += 1
            confirmed_pts = result.get("confirmed_osmosisPoints")
            print(f" ✅ (确认: {confirmed_pts})")
        else:
            err_count += 1
            print(f" ❌ {result.get('error', 'unknown')}")

print()
    print(f"✅ 完成！成功: {ok_count}, 失败: {err_count}")

# ── 主入口 ──────────────────────────────────────────────────────────────
def main() -> None:
    print("《即插即用》OS 自动重分配到恰好 10 万分")
    print()

credentials_path = _resolve_credentials_path()

while True:
        try:
            scope = input("请输入 scope（如 GLB/D0），直接回车退出: ").strip()
        except (EOFError, KeyboardInterrupt):
            print("\n已退出。")
            break

if not scope:
            print("已退出。")
            break

scope = scope.upper()
        print(f"\n── Scope: {scope} ──")
        process_scope(scope, credentials_path)
        print()

if __name__ == "__main__":
    main()
