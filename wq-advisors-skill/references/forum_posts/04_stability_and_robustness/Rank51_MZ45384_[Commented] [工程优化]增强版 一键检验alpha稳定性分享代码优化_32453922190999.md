### 评论 #4 (作者: XW23690, 时间: 8个月前)

感谢分享

近期平台回测总是失败，代码遇到网络波动容易报错。在这里代码进行下小小修改，增加重试

```
def fetch_locations(wqs, location_list):
    alpha_id_list = list()
    for location_url in location_list:
        while True:
            resp = wqs.get(location_url)
            logger.info(f'fetch location {location_url}')
            if 'progress' not in resp.text and 'error' not in resp.text:
                for children_id in resp.json()['children']:
                    max_retry = 3
                    retry_count = 0
                    child_json = None  

                    while retry_count < max_retry:
                        try:
                            child_resp = wqs.get(f'https://api.worldquantbrain.com/simulations/{children_id}')
                            child_json = child_resp.json()
                            break  
                        except json.JSONDecodeError:
                            retry_count += 1
                            wait_time = 2
                            logger.warning(f"子模拟 {children_id} JSON解析失败，第{retry_count}/{max_retry}次重试，等待{wait_time}秒")
                            time.sleep(wait_time)

                    # 重试后仍失败：跳过这个子模拟
                    if child_json is None:
                        logger.error(f"子模拟 {children_id} 重试{max_retry}次后仍失败，跳过")
                        continue

                    alpha_id = child_json.get("alpha")
                    if alpha_id: 
                        logger.info(f'获取子模拟 alpha_id: {alpha_id}')
                        alpha_id_list.append(alpha_id)
                    else:
                        logger.warning(f"子模拟 {children_id} 未找到 alpha_id，跳过")

                break
            else:
                logger.info("回测仍在进行中，60秒后重试...")
                time.sleep(60)
    return alpha_id_list
```
